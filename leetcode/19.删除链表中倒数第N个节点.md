/**
 * Definition for singly-linked list.
 * type ListNode struct {
 *     Val int
 *     Next *ListNode
 * }
 */
func removeNthFromEnd(head *ListNode, n int) *ListNode {
    // 快慢指针
	f,s:= head,head
	for i := 0; i < n; i++ {
		f = f.Next
	}
    // 如果f移动到尾指针的位置，说明要删除第一个节点
    if f == nil {
        return head.Next
    }
	for f.Next != nil {
		f = f.Next
		s = s.Next
	}
    
	tmp := s.Next
    // tmp 即为要删除的节点
	s.Next = tmp.Next
	return head
}
